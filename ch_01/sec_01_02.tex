\section{The Real Line and Coordinate Plane. Pythagoras}

\begin{questions}
    \item Among the words "integer", "rational", and "irrational", state the ones that apply to
        \begin{tasks}(4)
            \task \( -\frac{2}{3}\);
            \task \(0\);
            \task \(\frac{45}{9}\);
            \task \(0.75\);
            \task \(-\sqrt{49}\);
            \task \(\frac{1}{\pi}\);
            \task \(9.000...\);
            \task \(3^{1/2}\);
            \task \(-\frac{20}{7}\);
            \task \(\frac{94}{7}\).
        \end{tasks}
        \begin{note}
            \begin{itemize}
                \item \(\frac{45}{9} = 5\)
                \item \(-\sqrt{49} = -7\)
                \item \(9.000... = 9\)
            \end{itemize}
        \end{note}
        \begin{solution}
            \begin{itemize}
                \item integer: \(0\), \(\frac{45}{9}\), \(-\sqrt{49}\), \(9.000...\);
                \item rational: \(-\frac{2}{3}\), \(0.75\), \(-\frac{20}{7}\), \(\frac{94}{7}\);
                \item irrational: \(\frac{1}{\pi}\), \(3^{1/2}\).
            \end{itemize}
        \end{solution}
\end{questions}
